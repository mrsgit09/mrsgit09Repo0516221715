<p>The DISCOVER_XEVENT_OBJECTS<a id="Appendix_A_Target_317"></a><a href="b9ac4859-2662-44ca-b131-9addd8b953dc.htm#Appendix_A_317" aria-label="Product behavior note 317">&lt;317&gt;</a> rowset
contains the following columns.</p>

<table>
 <thead>
  <tr>
   <th>
   <p>Name</p>
   </th>
   <th>
   <p>Type</p>
   </th>
   <th>
   <p>Restriction</p>
   </th>
   <th>
   <p>Description</p>
   </th>
  </tr>
 </thead>
 <tr>
  <td>
  <p>NAME</p>
  </td>
  <td>
  <p>xsd:string</p>
  </td>
  <td>
  <p>Yes</p>
  </td>
  <td>
  <p>The name of the object. The name is unique within a
  package for a specific object type.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>OBJECT_TYPE</p>
  </td>
  <td>
  <p>xsd:string</p>
  </td>
  <td>
  <p>Yes</p>
  </td>
  <td>
  <p>The type of the object. Valid values are:</p>
  <p><span>§<span>  
  </span></span>Event</p>
  <p><span>§<span>  
  </span></span>Action</p>
  <p><span>§<span>  
  </span></span>Target</p>
  <p><span>§<span>  
  </span></span>pred_source</p>
  <p><span>§<span>  
  </span></span>pred_compare</p>
  <p><span>§<span>  
  </span></span>Type</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>PACKAGE_ID</p>
  </td>
  <td>
  <p>uuid</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>The unique identifier of the XEvent package that
  exposes this object, as a GUID.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>DESCRIPTION</p>
  </td>
  <td>
  <p>xsd:string</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>A description of the object.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>TYPE_NAME</p>
  </td>
  <td>
  <p>xsd:string</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>The name for pred_source and pred_compare objects.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>TYPE_PACKAGE_ID</p>
  </td>
  <td>
  <p>uuid</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>The unique identifier of the XEvent package that
  exposes the type of this object, as a GUID.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>TYPE_SIZE</p>
  </td>
  <td>
  <p>xsd:int</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>The size in bytes of the data type. This value is used
  only for valid object types.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>CAPABILITIES</p>
  </td>
  <td>
  <p>xsd:int</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>A bitmask that describes the capabilities of the
  object.</p>
  </td>
 </tr>
 <tr>
  <td>
  <p>CAPABILITIES_DESC</p>
  </td>
  <td>
  <p>xsd:string</p>
  </td>
  <td>
  <p> </p>
  </td>
  <td>
  <p>A list of all the capabilities of the object.</p>
  </td>
 </tr>
</table>
