<dd>
<div><pre> &lt;xsd:complexType name=&quot;PageBreakType&quot;&gt;

       &lt;xsd:simpleType&gt;
         &lt;xsd:restriction base=&quot;xsd:string&quot;&gt;

</pre></div>
