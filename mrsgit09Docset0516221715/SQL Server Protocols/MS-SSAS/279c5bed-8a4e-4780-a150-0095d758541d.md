columns.</p>

 <thead>
  <tr>
   <th>

   </th>
   <th>

 </thead>
 <tr>
  <td>

  </td>
  <td>
  <p>xsd:string</p>

 </tr>

  </span></span><span>Paused (0x00000007)</span></p>
  </li><li><p><span><span>  
  </span></span><span>Pause Pending (0x00000006)</span></p>
  </li><li><p><span><span>  
  </span></span><span>Continue Pending (0x00000005)</span></p>
  </li></ul></td>
 </tr>
